## 单选组

<Playground>
  {
    () => {
      const [val, setVal] = useState(3);

      const handleChange = e => {
        setVal(e.target.value);
      };

      return (
        <Radio.Group value={val} onChange={(e) => {handleChange(e)}}>
          <Radio value={3}>Radio 3</Radio>
          <Radio value={4}>Radio 4</Radio>
        </Radio.Group>
      )
    }
  }
</Playground>
